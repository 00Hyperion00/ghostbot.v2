def _q_dict(value):
    return value if isinstance(value, dict) else {}


def _q_list(value):
    if isinstance(value, list):
        return value
    if isinstance(value, tuple):
        return list(value)
    return []


def _q_bool(value):
    if isinstance(value, str):
        return value.strip().lower() in {'1', 'true', 'yes', 'on', 'enabled', 'ready', 'normal'}
    return bool(value)


def _q_float(value, default=0.0):
    try:
        if value in (None, ''):
            return default
        return float(value)
    except Exception:
        return default


def _q_int(value, default=0):
    try:
        if value in (None, ''):
            return default
        return int(float(value))
    except Exception:
        return default


def _q_fmt(value, digits=6):
    try:
        if value is None:
            return '-'
        return f'{float(value):.{int(digits)}f}'
    except Exception:
        return '-' if value in (None, '') else str(value)


def _q_is_all(value):
    if value is None:
        return True
    raw = str(value).strip()
    return raw in {'', '-', 'All', 'ALL', 'all', 'Tümü', 'TUMU', 'Tümü / All'}


def _q_getattr(obj, name, default=None):
    try:
        return object.__getattribute__(obj, name)
    except Exception:
        try:
            return getattr(obj, name)
        except Exception:
            return default


if 'safe_obj_getattr' not in globals():

def _q_var_value(obj, default=None):
    if obj is None:
        return default
    try:
        if hasattr(obj, 'get') and callable(obj.get):
            return obj.get()
    except Exception:
        pass
    return obj if obj is not None else default


def _q_write(app, target, text):
    setter = _q_getattr(app, '_set_text', None)
    if callable(setter):
        try:
            setter(target, text)
            return
        except Exception:
            pass
    if isinstance(target, str):
        widget = _q_getattr(app, target, None)
    else:
        widget = target
    if widget is None:
        return
    try:
        widget.text = text
    except Exception:
        pass
    try:
        widget.delete('1.0', 'end')
        widget.insert('end', text)
        return
    except Exception:
        pass
    try:
        widget.configure(text=text)
    except Exception:
        pass


def _q_cfg(widget, **kwargs):
    if widget is None:
        return
    try:
        widget.configure(**kwargs)
    except Exception:
        pass
    try:
        widget.kwargs.update(kwargs)
    except Exception:
        pass
    for key, value in kwargs.items():
        try:
            setattr(widget, key, value)
        except Exception:
            pass


def _q_widget_value(app, *names, default=None):
    for name in names:
        obj = _q_getattr(app, name, None)
        if obj is not None:
            value = _q_var_value(obj, None)
            if value is not None:
                return value
    return default


def _q_health_value(value):
    raw = str(value or 'HEALTHY').upper()
    if raw in {'HEALTHY', 'OK'}:
        return 'OK'
    if raw in {'WARNING', 'WARN'}:
        return 'WARN'
    return raw

def build_operator_control_state(status=None, *, connected=True, **kwargs):
    status = dict(status or {})
    status.update({k: v for k, v in kwargs.items() if v is not None and k not in status})
    health = _q_dict(status.get('health_snapshot'))
    risk = _q_dict(status.get('risk_snapshot') or status)
    pos = _q_position(status)
    prot = _q_dict(pos.get('protective_exit'))
    pending = _q_present_pending(status)
    in_position = _q_present_position(status)
    state = _q_state(status)
    safe_mode = _q_bool(risk.get('safe_mode') or status.get('safe_mode'))
    kill_switch = _q_bool(risk.get('kill_switch_active') or status.get('kill_switch_active'))
    contract_ok = _q_contract_ok(status.get('contract_version'))
    health_ok = True
    warnings = []
    reason_codes = []
    for key, code in [
        ('account_consistency', 'ACCOUNT_CONSISTENCY_BROKEN'),
        ('position_consistency', 'ACCOUNT_POSITION_DRIFT'),
        ('pending_consistency', 'PENDING_CONSISTENCY_BROKEN'),
    ]:
        raw = str(health.get(key, 'HEALTHY') or 'HEALTHY').upper()
        if raw not in {'HEALTHY', 'OK'}:
            health_ok = False
            warnings.append(code)
            reason_codes.append(f'HEALTH_ANOMALY:{code}')
    active_anomaly = health.get('active_anomaly_code')
    if active_anomaly:
        health_ok = False
        warnings.append(str(active_anomaly))
        reason_codes.append(f'HEALTH_ANOMALY:{active_anomaly}')
    if not reason_codes and not health_ok:
        reason_codes.append('HEALTH_ANOMALY')

    if safe_mode:
        reason_codes.append('SAFE_MODE_ACTIVE')
    if kill_switch:
        reason_codes.append('KILL_SWITCH_ACTIVE')
    if not contract_ok:
        reason_codes.append('STATUS_CONTRACT_STALE')
    if pending:
        reason_codes.append('PENDING_ORDER_ACTIVE')

    is_dust = _q_bool(prot.get('is_dust'))
    if is_dust:
        reason_codes.append('POSITION_DUST')
    block_reason = prot.get('block_reason')
    protective_blocked = bool(block_reason and str(block_reason).upper() not in {'-', 'NONE', 'OK', ''})
    if protective_blocked:
        reason_codes.append(f'PROTECTIVE_EXIT_BLOCKED:{str(block_reason).upper()}')

    base_ok = bool(connected and contract_ok and health_ok and not kill_switch)
    force_buy = base_ok and not safe_mode and not pending and not in_position
    force_sell = base_ok and in_position and not pending and not protective_blocked and not is_dust
    cancel_pending = bool(connected and pending)
    stop = bool(connected)
    start = bool(connected and not status.get('running', True))

    if state == 'BUY_PENDING':
        hint = 'giriş emri bekliyor'
    elif state == 'SELL_PENDING':
        hint = 'çıkış emri bekliyor'
    elif pending:
        hint = 'pending emir var'
    elif safe_mode:
        hint = 'safe mode aktif'
    elif force_sell:
        hint = 'force sell aktif'
    elif force_buy:
        hint = 'Force BUY aktif'
    elif not health_ok:
        hint = ','.join(warnings) or 'health anomaly'
    elif not contract_ok:
        hint = 'status contract stale'
    elif protective_blocked:
        hint = str(block_reason)
    elif is_dust:
        hint = 'position dust'
    else:
        hint = 'READY'
    severity = 'ready'
    if pending:
        severity = 'busy'
    elif safe_mode:
        severity = 'safe'
    elif in_position:
        severity = 'position'
    if kill_switch or not health_ok or not contract_ok or protective_blocked:
        severity = 'danger'

    buttons = {
        'force_buy': force_buy,
        'force_sell': force_sell,
        'cancel_pending': cancel_pending,
        'safe_mode': bool(connected),
        'safe_mode_toggle': bool(connected),
        'balance_sync': bool(connected),
        'ai_reload': bool(connected),
        'start': start,
        'stop': stop,
    }
    return {
        'contract_version': DASHBOARD_CONTROL_CONTRACT_VERSION,
        'state': state,
        'connected': bool(connected),
        'health_ok': health_ok,
        'contract_ok': contract_ok,
        'safe_mode': safe_mode,
        'kill_switch_active': kill_switch,
        'has_position': in_position,
        'has_pending': pending,
        'position_is_dust': is_dust,
        'protective_exit_ready': bool(in_position and not protective_blocked and not is_dust),
        'severity': severity,
        'warnings': warnings,
        'reason_codes': list(dict.fromkeys(reason_codes)),
        'hint': hint,
        'buttons': buttons,
        **buttons,
    }

def _q_trade_items(logs):
    out = []
    for item in _q_list(logs):
        if isinstance(item, dict) and str(item.get('code', '')).upper() == 'POSITION_CLOSED':
            data = _q_dict(item.get('data'))
            out.append({'symbol': data.get('symbol') or item.get('symbol') or 'ETHUSDT', 'pnl': _q_float(data.get('pnl', item.get('pnl')), 0.0), 'ts': _q_float(item.get('ts'), 0.0)})
    return out


def _q_trade_text(trades):
    return ' / '.join(f"{t.get('symbol') or 'ETHUSDT'} {_q_float(t.get('pnl')):.6f}" for t in trades) or '-'


def _q_wlbe(trades):
    w = l = b = 0
    for t in trades:
        pnl = _q_float(t.get('pnl'))
        if pnl > 1e-12:
            w += 1
        elif pnl < -1e-12:
            l += 1
        else:
            b += 1
    return w, l, b


def _render_session_summary_20q(self, status=None):
    status = _q_dict(status or _q_getattr(self, '_last_status', {}))
    logs = _q_list(_q_getattr(self, '_log_items', None) or _q_getattr(self, '_last_logs', None))
    trades_all = _q_trade_items(logs)
    reset_ts = None
    for item in logs:
        if isinstance(item, dict) and str(item.get('code', '')).upper() == 'RISK_STATS_RESET':
            reset_ts = _q_float(item.get('ts'))
    if reset_ts is not None:
        scoped = [t for t in trades_all if _q_float(t.get('ts')) > reset_ts]
    else:
        scoped = list(trades_all)
    risk = _q_dict(status.get('risk_snapshot'))
    session = _q_dict(status.get('session'))
    daily_count = _q_int(session.get('daily_trade_count', risk.get('daily_trade_count', len(scoped))))
    w, l, b = _q_wlbe(scoped)
    tracked_pnl = sum(_q_float(t.get('pnl')) for t in scoped)
    today_mode = reset_ts is not None or daily_count == len(scoped)
    wl_label = 'Today W/L/BE' if today_mode else 'Tracked W/L/BE'
    trade_label = 'Today trades' if today_mode else 'Tracked trades'
    warnings = [i for i in logs if isinstance(i, dict) and str(i.get('level', '')).upper() in {'WARN', 'WARNING', 'ERROR', 'CRITICAL'}]
    ai = _q_dict(status.get('ai_snapshot'))
    health = _q_dict(status.get('health_snapshot'))
    health_line = '/'.join(_q_health_value(health.get(k)) for k in ('account_consistency', 'position_consistency', 'pending_consistency'))
    recent = trades_all[-3:]
    scope_note = '-' if today_mode and daily_count == len(scoped) else f'partial log scope ({len(scoped)}/{daily_count})'
    lines = [
        f"Current signal  : {status.get('last_signal', '-')}",
        f"Signal reason   : {status.get('signal_reason', '-')}",
        f"Trend           : {status.get('trend', '-')}",
        f"Tracked PnL     : {tracked_pnl:.6f}",
        f"Trades today    : {daily_count}",
        f"{wl_label:<16}: {w}/{l}/{b}",
        f"{trade_label:<16}: {_q_trade_text(scoped)}",
    ]
    if today_mode:
        lines.append(f"Recent hist.    : {_q_trade_text(recent)}")
    lines.extend([
        f"Last warning    : {warnings[-1].get('code') if warnings else '-'}",
        f"Health          : {health_line}",
        f"Model           : {ai.get('model_path', '-')}",
        f"Confidence      : {'-' if ai.get('confidence') is None else '%' + format(_q_float(ai.get('confidence')) * 100, '.1f')}",
        f"Scope note      : {scope_note}",
    ])
    _q_write(self, _q_getattr(self, 'log_box', 'log-box'), '\n'.join(lines))

def _q_event_category(item):
    item = _q_dict(item)
    if item.get('category'):
        return str(item.get('category'))
    code = str(item.get('code') or '').upper()
    if code.startswith(('ORDER_', 'POSITION_', 'LIVE_', 'FILL_')):
        return 'Orders'
    if code.startswith(('AUTO_', 'ENTRY_GUARD', 'EXIT_GUARD')):
        return 'Guards'
    if code.startswith(('SAFE_', 'RISK_', 'KILL_')):
        return 'Warnings'
    if code.startswith(('AI_', 'MODEL_', 'STRATEGY_')):
        return 'AI'
    return 'System'


def _q_event_severity(item):
    item = _q_dict(item)
    if item.get('severity'):
        return str(item.get('severity')).lower()
    level = str(item.get('level') or '').upper()
    if level in {'WARN', 'WARNING'}:
        return 'warning'
    if level in {'ERROR', 'CRITICAL'}:
        return 'error'
    code = str(item.get('code') or '').upper()
    if code.startswith('AUTO_') or code.startswith('SAFE_'):
        return 'warning'
    return 'info'


def _q_corr(item):
    item = _q_dict(item)
    data = _q_dict(item.get('data'))
    for key in ('correlation_id', 'correlationId', 'clientOrderId', 'client_order_id', 'orderId', 'order_id'):
        if item.get(key):
            return str(item.get(key))
        if data.get(key):
            return str(data.get(key))
    return '-'


def _q_blob(item):
    item = _q_dict(item)
    try:
        return ' '.join([str(item.get('code') or ''), str(item.get('message') or ''), _q_event_category(item), _q_event_severity(item), _q_corr(item), _q_json.dumps(item.get('data') or {}, ensure_ascii=False)]).lower()
    except Exception:
        return str(item).lower()


def format_log_line(item):
    item = _q_dict(item)
    return f"{item.get('ts', '-')} | {str(item.get('level') or 'INFO').upper()} | {_q_event_category(item)} | {_q_event_severity(item)} | {item.get('code') or '-'} | corr={_q_corr(item)} | {item.get('message') or ''} | {_q_dict(item.get('data'))}"


def build_audit_query_path(*, limit=50, order='desc', level=None, code=None, code_prefix=None, category=None, severity=None, contains=None, q=None, correlation=None, **extra):
    params = {'limit': int(limit), 'order': str(order or 'desc').lower()}
    if not _q_is_all(level):
        params['level'] = str(level).strip().upper().replace('WARNING', 'WARN')
    if not _q_is_all(code):
        params['code'] = str(code).strip().upper()
    if not _q_is_all(code_prefix):
        params['code_prefix'] = str(code_prefix).strip().upper()
    if not _q_is_all(category):
        params['category'] = str(category).strip()
    if not _q_is_all(severity):
        params['severity'] = str(severity).strip().lower()
    if not _q_is_all(correlation):
        params['correlation'] = str(correlation).strip()
    query = q if not _q_is_all(q) else contains
    if not _q_is_all(query):
        params['q'] = str(query).strip()
    return '/events/audit?' + _q_urlencode(params)


def filter_audit_events(events, filters=None, **kwargs):
    merged = {}
    if isinstance(filters, dict):
        merged.update(filters)
    merged.update({k: v for k, v in kwargs.items() if v is not None})
    out = [dict(i) for i in _q_list(events) if isinstance(i, dict)]
    if not _q_is_all(merged.get('category')):
        wanted = str(merged.get('category')).lower()
        out = [i for i in out if _q_event_category(i).lower() == wanted]
    if not _q_is_all(merged.get('severity')):
        wanted = str(merged.get('severity')).lower()
        out = [i for i in out if _q_event_severity(i) == wanted]
    if not _q_is_all(merged.get('code_prefix')):
        pref = str(merged.get('code_prefix')).upper()
        out = [i for i in out if str(i.get('code') or '').upper().startswith(pref)]
    if not _q_is_all(merged.get('code')):
        code = str(merged.get('code')).upper()
        out = [i for i in out if str(i.get('code') or '').upper() == code]
    if not _q_is_all(merged.get('correlation')):
        corr = str(merged.get('correlation')).lower()
        out = [i for i in out if corr in _q_corr(i).lower() or corr in _q_blob(i)]
    query = merged.get('q') if not _q_is_all(merged.get('q')) else merged.get('contains')
    if not _q_is_all(query):
        needle = str(query).lower()
        out = [i for i in out if needle in _q_blob(i)]
    out.sort(key=lambda i: _q_float(i.get('ts')), reverse=str(merged.get('order', 'desc')).lower() != 'asc')
    return out


def _q_counts_from_payload(payload, events):
    payload = _q_dict(payload)
    summary = _q_dict(payload.get('summary') or payload.get('counts') or payload.get('event_summary'))
    categories = _q_dict(payload.get('categories') or payload.get('category_counts') or summary.get('categories') or summary.get('category_counts'))
    severities = _q_dict(payload.get('severities') or payload.get('severity_counts') or summary.get('severities') or summary.get('severity_counts'))
    codes = _q_dict(payload.get('codes') or payload.get('code_counts') or summary.get('codes') or summary.get('code_counts'))
    if not categories:
        for i in events:
            c = _q_event_category(i); categories[c] = categories.get(c, 0) + 1
    if not severities:
        for i in events:
            s = _q_event_severity(i); severities[s] = severities.get(s, 0) + 1
    if not codes:
        for i in events:
            c = str(i.get('code') or '-'); codes[c] = codes.get(c, 0) + 1
    return categories, severities, codes


def build_audit_summary_text(payload=None, logs=None):
    if isinstance(payload, list):
        events = [dict(i) for i in payload if isinstance(i, dict)]
        data = {}
    else:
        data = _q_dict(payload)
        raw = logs if logs is not None else data.get('events') or data.get('items') or data.get('logs') or data.get('filtered_events') or []
        events = [dict(i) for i in _q_list(raw) if isinstance(i, dict)]
    categories, severities, codes = _q_counts_from_payload(data, events)
    total = _q_int(data.get('total', data.get('total_events', sum(categories.values()) or len(events))))
    warn_count = int(severities.get('warning', 0) or severities.get('WARN', 0) or 0)
    err_count = int(severities.get('error', 0) or severities.get('ERROR', 0) or 0)
    fmt = lambda d: ', '.join(f'{k}:{v}' for k, v in sorted(d.items())) if d else '-'
    return '\n'.join([
        'Audit Viewer',
        '------------',
        f'Contract        : {AUDIT_VIEWER_CONTRACT_VERSION}',
        f'Total events    : {total}',
        f'Rendered count  : {len(events) if events else total}',
        f'Filtered events : {len(events) if events else total}',
        f'Warnings/errors : {warn_count} / {err_count}',
        f'Categories      : {fmt(categories)}',
        f'Severities      : {fmt(severities)}',
        f'Codes           : {fmt(codes)}',
        f'Top codes       : {fmt(codes)}',
    ])


def _q_collect_events(app):
    for name in ('_audit_events', '_last_audit_events', '_log_items', '_last_logs', 'logs'):
        value = _q_getattr(app, name, None)
        if isinstance(value, list):
            return [dict(i) for i in value if isinstance(i, dict)]
    getter = _q_getattr(app, 'api_get', None)
    if callable(getter):
        try:
            payload = getter('/events/audit', timeout=2.0)
            if isinstance(payload, dict):
                return [dict(i) for i in _q_list(payload.get('events') or payload.get('items') or payload.get('logs')) if isinstance(i, dict)]
        except Exception:
            pass
    return []


def _render_logs_20q(self, payload=None):
    events = _q_collect_events(self) if payload is None else [dict(i) for i in _q_list(_q_dict(payload).get('events') or _q_dict(payload).get('items') or _q_dict(payload).get('logs')) if isinstance(i, dict)]
    filters = {
        'category': _q_widget_value(self, 'audit_category_var', 'audit_category_filter', default='All'),
        'severity': _q_widget_value(self, 'audit_severity_var', 'audit_severity_filter', default='All'),
        'code_prefix': _q_widget_value(self, 'audit_code_prefix_var', 'audit_code_prefix_filter', default=None),

        'correlation': _q_widget_value(self, 'audit_correlation_var', 'audit_correlation_filter', default=None),

def _extract_training_output_path_20q(self, line):
    raw = str(line or '').strip()
    if not raw:
        return None
    try:
        obj = _q_json.loads(raw)
    except Exception:
        try:
            obj = _q_ast.literal_eval(raw)
        except Exception:
            obj = None
    if isinstance(obj, dict):
        for key in ('model_path', 'output', 'output_path', 'path'):
            if obj.get(key):
                return str(obj.get(key))
    for marker in ('model_path=', 'output_path=', 'output='):
        if marker in raw:
            return raw.split(marker, 1)[1].strip().strip('"\'').split()[0].strip(',;')
    return None
